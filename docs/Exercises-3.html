3.37 It is difficult to compute N! when N is large. As an approximation, we can use Stirling's formula, which says that for large N

\begin{equation*} N! \approx \sqrt{2\pi}N^{N+1/2}e^{-N} \end{equation*}

def log_stirlings_formaula(N):
    return 0.5*np.log(2*np.pi) + (N+0.5)*np.log(N) - N

def log_N_factorial(N):
    return np.sum(np.log(np.arange(1, N+1)))

fig, (ax1,ax2) = plt.subplots(2, 1, sharex=False, sharey=False, figsize=(6,10))

x = np.arange(1,300)
ax1.plot(x, [log_stirlings_formaula(n) for n in x], label='stirling')
ax1.plot(x, [log_N_factorial(n) for n in x], '--', alpha=0.5, label='factorial')
ax1.legend()
ax1.set_title('Log comparison of Factorial and Stirlings Formula')

ax2.plot(x, [np.e**(log_stirlings_formaula(n) - log_N_factorial(n)) for n in x])
ax2.set_title('stirling(n) - n!')
